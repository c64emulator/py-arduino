Example: blink led

/*
 * Blink LED on PIN 13
 */

var pinToBlink = 13;

logMessage("Setting pin " + pinToBlink + " as OUTPUT");
proxy.pinMode(pinToBlink, OUTPUT);

if(! confirm("Start to blink?")) throw "stopping script";

logMessage("Starting to blink");
var i;
for(i=0; i<5; i++) {
    proxy.digitalWrite(pinToBlink, HIGH);
    proxy.delay(250);
    proxy.digitalWrite(pinToBlink, LOW);
    proxy.delay(250);
}

Example: read temperature with LM335

/***********************************************************************
 * Second exampe. Using javascript functions.
 ***********************************************************************/

function readTemp(analogPort) {
    var value = proxy.analogRead(analogPort);
    var volt = (5.0/1024.0) * value;
    var kelvin = volt * 100;
    return (kelvin - 273); // return temp. in celcius
}

var analogPort = 0;

logMessage("Temp. on A" + analogPort + ": " + readTemp(analogPort));

/***********************************************************************
 * Use 'toFixed()' to get a better formatting
 ***********************************************************************/

logMessage("Temp. on A" + analogPort + ": " + readTemp(analogPort).toFixed(1) + "º C");
